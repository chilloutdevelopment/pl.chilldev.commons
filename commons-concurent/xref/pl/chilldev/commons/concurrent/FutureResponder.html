13  package pl.chilldev.commons.concurrent;
14  
15  import java.util.concurrent.Callable;
16  import java.util.concurrent.FutureTask;
17  
18  /**
19   * Generic future objects wrapper.
20   *
21   * @param <ResponseType> Any type that is expected to be return for your listener.
22   * @version 0.0.1
23   * @since 0.0.1
24   */
25  public class FutureResponder<ResponseType> implements Callable<ResponseType>
26  {
27      /**
28       * Bound future.
29       */
30      protected FutureTask future;
31  
32      /**
33       * Response to return.
34       */
35      protected ResponseType response;
36  
37      /**
38       * Sets future to notify.
39       *
40       * @param future Target future.
41       * @return Self instance.
42       * @since 0.0.1
43       */
44      public FutureResponder<ResponseType> setFuture(FutureTask future)
45      {
46          this.future = future;
47  
48          return this;
49      }
50  
51      /**
52       * Assigns response.
53       *
54       * @param response Response to return.
55       * @return Self instance.
56       * @since 0.0.1
57       */
58      public FutureResponder<ResponseType> setResponse(ResponseType response)
59      {
60          this.response = response;
61          this.future.run();
62  
63          return this;
64      }
65  
66      /**
67       * Returns previously-sasigned response.
68       *
69       * @return Pre-defined response.
70       * @since 0.0.1
71       */
72      public ResponseType call()
73      {
74          return this.response;
75      }
76  }
