8   package pl.wrzasq.commons.aws.cloudformation;
9   
10  import java.util.Collection;
11  import java.util.Collections;
12  import java.util.Map;
13  import java.util.Optional;
14  import java.util.function.BiFunction;
15  import java.util.stream.Collectors;
16  
17  /**
18   * Custom utility routines for dealing with CloudFormation data operations.
19   */
20  public class StackUtils {
21      /**
22       * Converts key-value mapping into SDK data structure.
23       *
24       * @param input Key-value plain values.
25       * @param converter Converter for single key-value tuple into SDK structure.
26       * @param <Type> SDK element type.
27       * @return List of SDK elements.
28       */
29      public static <Type> Collection<Type> buildSdkList(
30          Map<String, String> input,
31          BiFunction<String, String, Type> converter
32      ) {
33          return Optional.ofNullable(input)
34              .orElse(Collections.emptyMap())
35              .entrySet()
36              .stream()
37              .map(entry -> converter.apply(entry.getKey(), entry.getValue()))
38              .collect(Collectors.toList());
39      }
40  }
