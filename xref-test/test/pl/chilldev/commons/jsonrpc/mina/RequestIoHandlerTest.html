36  @RunWith(MockitoJUnitRunner.class)
37  public class RequestIoHandlerTest

39      @Mock

41  
42      @Captor
43      private ArgumentCaptor<Object> captor;
44  
45      @Test(expected = TimeoutException.class)
46      public void sessionIdle()

48              TimeoutException

50          ProtocolCodecSession session = new ProtocolCodecSession();
51          RequestIoHandler handler = new RequestIoHandler();

56      @Test(expected = Exception.class)
57      public void exceptionCaught()
58          throws
59              Exception
60      {
61          ProtocolCodecSession session = new ProtocolCodecSession();
62          RequestIoHandler handler = new RequestIoHandler();
63  
64          handler.exceptionCaught(session, new Exception("test"));
65      }
66  
67      @Test
68      public void execute()
69          throws
70              Exception
71      {
72          ProtocolCodecSession session = new ProtocolCodecSession();
73          RequestIoHandler handler = new RequestIoHandler();
74  
75          session.setHandler(this.handler);
76  
77          handler.sessionOpened(session);
78  
79          FutureTask<JSONRPC2Response> future = handler.execute("test");
80  
81          verify(this.handler).messageSent(same(session), this.captor.capture());
82  
83          String json = this.captor.getValue().toString();
84          JSONRPC2Request request = JSONRPC2Request.parse(json);
85          String id = request.getID().toString();
86          JSONAssert.assertEquals("{\"id\":" + id + ",\"method\":\"test\",\"jsonrpc\":\"2.0\"}", json, true);
87  
88          handler.messageReceived(session, "{\"jsonrpc\":\"2.0\",\"id\":" + id + ",\"result\":\"foo\"}");
89  
90          assertEquals(
91              "RequestIoHandler.messageReceived() should dispatch result to associated pending request future.",
92              "foo",
93              future.get().getResult()
94          );
95      }
96  
97      @Test
98      public void executeWithParams()
99          throws
100             Exception
101     {
102         ProtocolCodecSession session = new ProtocolCodecSession();
103         RequestIoHandler handler = new RequestIoHandler();
104 
105         session.setHandler(this.handler);
106 
107         handler.sessionOpened(session);
108 
109         Map<String, Object> params = new HashMap<>();
110         params.put("id", 123);
111         FutureTask<JSONRPC2Response> future = handler.execute("test", params);
112 
113         verify(this.handler).messageSent(same(session), this.captor.capture());
114 
115         String json = this.captor.getValue().toString();
116         JSONRPC2Request request = JSONRPC2Request.parse(json);
117         String id = request.getID().toString();
118         JSONAssert.assertEquals("{\"id\":" + id + ",\"method\":\"test\",\"params\":{\"id\":123},\"jsonrpc\":\"2.0\"}", json, true);
119 
120         handler.messageReceived(session, "{\"jsonrpc\":\"2.0\",\"id\":" + id + ",\"result\":\"foo\"}");
121 
122         assertEquals(
123             "RequestIoHandler.messageReceived() should dispatch result to associated pending request future.",
124             "foo",
125             future.get().getResult()
126         );
127     }
