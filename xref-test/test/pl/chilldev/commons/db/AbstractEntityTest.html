10  import java.util.UUID;
11  
12  import org.junit.Test;
13  import org.junit.Assert;
14  
15  import pl.chilldev.commons.db.AbstractEntity;
16  
17  public class AbstractEntityTest
18  {
19      @Test
20      public void getId()
21      {
22          AbstractEntity entity = new Entity();
23  
24          Assert.assertNull(
25              "AbstractEntity.getId() should return NULL if entity is not persisted.",
26              entity.getId()
27          );
28      }
29  
30      @Test
31      public void setId()
32      {
33          UUID id = UUID.randomUUID();
34          AbstractEntity entity = new Entity();
35          entity.setId(id);
36  
37          Assert.assertEquals(
38              "AbstractEntity.getId() should return previously set ID if assigned.",
39              id,
40              entity.getId()
41          );
42      }
43  
44      @Test
45      public void equalsSame()
46      {
47          AbstractEntity entity = new Entity();
48          Assert.assertTrue(
49              "AbstractEntity.equals() should return TRUE if the object is the very same instance.",
50              entity.equals(entity)
51          );
52      }
53  
54      @Test
55      public void equalsNull()
56      {
57          AbstractEntity entity = new Entity();
58          Assert.assertFalse(
59              "AbstractEntity.equals() should return FALSE when compared to NULL.",
60              entity.equals(null)
61          );
62      }
63  
64      @Test
65      public void equalsNotEntity()
66      {
67          UUID id = UUID.randomUUID();
68          AbstractEntity entity = new Entity();
69          Assert.assertFalse(
70              "AbstractEntity.equals() should return FALSE when compared to a non-entity object.",
71              entity.equals(id)
72          );
73      }
74  
75      @Test
76      public void equalsNullId()
77      {
78          AbstractEntity object = new Entity();
79          AbstractEntity entity = new Entity();
80          Assert.assertFalse(
81              "AbstractEntity.equals() should return FALSE when it is not persisted.",
82              entity.equals(object)
83          );
84      }
85  
86      @Test
87      public void equals()
88      {
89          UUID id = UUID.randomUUID();
90  
91          AbstractEntity same = new Entity();
92          same.setId(id);
93  
94          AbstractEntity other = new Entity();
95          other.setId(UUID.randomUUID());
96  
97          AbstractEntity entity = new Entity();
98          entity.setId(id);
99  
100         Assert.assertTrue(
101             "AbstractEntity.equals() should return TRUE when compared entity has same ID.",
102             entity.equals(same)
103         );
104 
105         Assert.assertFalse(
106             "AbstractEntity.equals() should return FALSE when compared entity has different ID.",
107             entity.equals(other)
108         );
109     }
110 
111     @Test
112     public void testHashCode()
113     {
114         UUID id = UUID.randomUUID();
115         AbstractEntity entity = new Entity();
116         entity.setId(id);
117 
118         Assert.assertEquals(
119             "AbstractEntity.hashCode() should return hash code that is equal to it's ID hash code.",
120             id.hashCode(),
121             entity.hashCode()
122         );
123     }
124 
125     @Test
126     public void testHashCodeNull()
127     {
128         AbstractEntity entity = new Entity();
129 
130         Assert.assertEquals(
131             "AbstractEntity.hashCode() should return 0 if entity is not persisted.",
132             0,
133             entity.hashCode()
134         );
135     }
136 }
