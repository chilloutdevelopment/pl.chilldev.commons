8   package pl.chilldev.commons.jsonrpc.rpc;
9   
10  import java.util.Collections;
11  import java.util.HashSet;
12  import java.util.List;
13  import java.util.Set;
14  import java.util.UUID;
15  
16  import pl.chilldev.commons.jsonrpc.json.ParamsRetriever;
17  import pl.chilldev.commons.jsonrpc.rpc.introspector.Introspector;
18  
19  /**
20   * Core Java types handling.
21   */
22  public class JavaModule implements DispatcherModule
23  {
24      /**
25       * {@inheritDoc}
26       */
27      @Override
28      public void initializeIntrospector(Introspector introspector)
29      {
30          // parameters retrievers
31  
32          // boolean retriever
33          introspector.registerParameterProvider(
34              boolean.class,
35              (String name, ParamsRetriever params, boolean optional, String defaultValue) ->
36                  optional
37                      ? params.getOptBoolean(name, "true".equalsIgnoreCase(defaultValue))
38                      : params.getBoolean(name)
39          );
40  
41          // integer retriever
42          introspector.registerParameterProvider(
43              int.class,
44              (String name, ParamsRetriever params, boolean optional, String defaultValue) ->
45                  optional
46                      ? params.getOptInt(name, Integer.parseInt(defaultValue))
47                      : params.getInt(name)
48          );
49  
50          // long retriever
51          introspector.registerParameterProvider(
52              long.class,
53              (String name, ParamsRetriever params, boolean optional, String defaultValue) ->
54                  optional
55                      ? params.getOptLong(name, Long.parseLong(defaultValue))
56                      : params.getLong(name)
57          );
58  
59          // string retriever
60          introspector.registerParameterProvider(
61              String.class,
62              (String name, ParamsRetriever params, boolean optional, String defaultValue) ->
63                  optional
64                      ? params.getOptString(name, defaultValue)
65                      : params.getString(name)
66          );
67  
68          // UUID retriever
69          introspector.registerParameterProvider(
70              UUID.class,
71              (String name, ParamsRetriever params, boolean optional, String defaultValue) ->
72                  optional
73                      ? params.getOptUuid(name)
74                      : params.getUuid(name)
75          );
76  
77          // List retriever
78          introspector.registerParameterProvider(
79              List.class,
80              (String name, ParamsRetriever params, boolean optional, String defaultValue) ->
81                  optional
82                      ? params.getOptList(name, Collections.singletonList(defaultValue))
83                      : params.getList(name)
84          );
85  
86          // Set retriever
87          introspector.registerParameterProvider(
88              Set.class,
89              (String name, ParamsRetriever params, boolean optional, String defaultValue) ->
90                  new HashSet<>(
91                      optional
92                          ? params.getOptList(name, Collections.singletonList(defaultValue))
93                          : params.getList(name)
94                  )
95          );
96      }
97  }
