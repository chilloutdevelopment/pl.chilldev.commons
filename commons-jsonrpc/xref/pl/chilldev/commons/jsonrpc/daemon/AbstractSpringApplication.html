8   package pl.chilldev.commons.jsonrpc.daemon;
9   
10  import java.util.ArrayList;
11  import java.util.Collection;
12  
13  import org.springframework.beans.factory.BeanFactoryUtils;
14  
15  import org.springframework.context.annotation.AnnotationConfigApplicationContext;
16  
17  /**
18   * Base daemon implementation based on Spring context.
19   */
20  public abstract class AbstractSpringApplication extends AbstractApplication
21  {
22      /**
23       * Application context environment.
24       */
25      private AnnotationConfigApplicationContext context;
26  
27      /**
28       * Runs all listeners.
29       */
30      @Override
31      public void start()
32      {
33          // load fresh context
34          this.logger.info("Loading application context.");
35          this.context = new AnnotationConfigApplicationContext();
36          this.context.scan(this.getPackageToScan());
37          this.context.refresh();
38  
39          super.start();
40      }
41  
42      /**
43       * {@inheritDoc}
44       */
45      @Override
46      public void stop()
47      {
48          super.stop();
49  
50          // clean up the context
51          if (this.context != null) {
52              this.context.close();
53          }
54      }
55  
56      /**
57       * {@inheritDoc}
58       */
59      @Override
60      protected Collection<Listener<?>> buildListeners()
61      {
62          Collection<Listener<?>> listeners = new ArrayList<>();
63  
64          // this is to cast raw types into generics
65          BeanFactoryUtils.beansOfTypeIncludingAncestors(
66              this.context,
67              Listener.class
68          ).values().forEach(listeners::add);
69  
70          return listeners;
71      }
72  
73      /**
74       * Returns base package name for context scanning.
75       *
76       * @return Package name that contains any Spring configuration.
77       */
78      protected String getPackageToScan()
79      {
80          return this.getClass().getPackage().getName();
81      }
82  }
