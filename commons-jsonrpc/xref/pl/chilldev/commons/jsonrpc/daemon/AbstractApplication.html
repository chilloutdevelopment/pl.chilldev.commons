8   package pl.chilldev.commons.jsonrpc.daemon;
9   
10  import java.util.Collection;
11  import java.util.HashSet;
12  
13  import org.apache.commons.daemon.Daemon;
14  import org.apache.commons.daemon.DaemonContext;
15  import org.apache.commons.daemon.DaemonUserSignal;
16  
17  import org.slf4j.Logger;
18  import org.slf4j.LoggerFactory;
19  
20  /**
21   * Base daemon application class.
22   */
23  public abstract class AbstractApplication
24      implements
25          Daemon,
26          DaemonUserSignal
27  {
28      /**
29       * Logger.
30       */
31      protected Logger logger = LoggerFactory.getLogger(AbstractApplication.class);
32  
33      /**
34       * List of children threads.
35       */
36      protected Collection<Listener> threads = new HashSet<>();
37  
38      /**
39       * Runs all listeners.
40       */
41      @Override
42      public void start()
43      {
44          // running threads
45          for (Listener thread : this.buildListeners()) {
46              try {
47                  thread.start();
48                  this.threads.add(thread);
49              //CHECKSTYLE:OFF: IllegalCatchCheck
50              } catch (Throwable error) {
51              //CHECKSTYLE:ON: IllegalCatchCheck
52                  this.logger.error(
53                      "Not starting \"{}\" because of {}.",
54                      thread.getName(),
55                      error.getMessage(),
56                      error
57                  );
58              }
59          }
60  
61          this.logger.trace("Working.");
62      }
63  
64      /**
65       * Stops listeners.
66       */
67      @Override
68      public void stop()
69      {
70          this.logger.info("Stopping…");
71  
72          // stop all threads
73          this.threads.forEach(Listener::release);
74  
75          // wait for all threads
76          for (Thread thread : this.threads) {
77              try {
78                  // this is to make sure that thread don't hang on some I/O
79                  thread.interrupt();
80                  thread.join();
81              } catch (InterruptedException error) {
82                  this.logger.error("Had to interrupt while waiting for thread \"{}\".", thread.getName(), error);
83              }
84          }
85          // just to clean references
86          this.threads.clear();
87      }
88  
89      /**
90       * Just re-loads configuration.
91       */
92      @Override
93      public void signal()
94      {
95          this.logger.info("Handling configuration reload.");
96  
97          // no deep in-sight for now, but at least JVM bootstrapping is avoided
98  
99          this.stop();
100         this.start();
101     }
102 
103     /**
104      * Daemon resources initialization.
105      *
106      * @param context Runtime context for daemon.
107      */
108     @Override
109     public void init(DaemonContext context)
110     {
111         // diagnostic info
112         this.logger.info(
113             "Started {} daemon v.{} on {} {} ({}) / Java {} ({}).",
114             new Object[] {
115                 this.getDaemonName(),
116                 this.getDaemonVersion(),
117                 System.getProperty("os.name"),
118                 System.getProperty("os.version"),
119                 System.getProperty("os.arch"),
120                 System.getProperty("java.version"),
121                 System.getProperty("java.vendor"),
122             }
123         );
124     }
125 
126     /**
127      * Daemon resources freeing.
128      */
129     @Override
130     public void destroy()
131     {
132         // dummy method; not used - needed by org.apache.commons.daemon.Daemon
133     }
134 
135     /**
136      * Returns daemon application name.
137      *
138      * @return String representation for information purpose.
139      */
140     protected abstract String getDaemonName();
141 
142     /**
143      * Returns daemon application version.
144      *
145      * @return String representation for information purpose.
146      */
147     protected abstract String getDaemonVersion();
148 
149     /**
150      * Retruns collection of listeners (not started, but instantiated).
151      *
152      * @return Collection of listeners to be started by the daemon.
153      */
154     protected abstract Collection<Listener> buildListeners();
155 }
