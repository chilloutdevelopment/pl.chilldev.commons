8   package pl.chilldev.commons.client.interceptor;
9   
10  import feign.RequestInterceptor;
11  import feign.RequestTemplate;
12  import org.springframework.http.HttpHeaders;
13  
14  /**
15   * Feign request interceptor that injects custom HTTP authorization with Bearer token.
16   */
17  public class BearerTokenAuthorizer implements RequestInterceptor {
18      /**
19       * Bearer token prefix.
20       */
21      private static final String TYPE_BEARER = "Bearer ";
22      /**
23       * HTTP access token..
24       */
25      private String token;
26  
27      /**
28       * {@inheritDoc}
29       */
30      @Override
31      public void apply(RequestTemplate requestTemplate) {
32          // set the same value as incoming request to execute requests on behalf of the user
33          requestTemplate.header(HttpHeaders.AUTHORIZATION, BearerTokenAuthorizer.TYPE_BEARER + this.token);
34      }
35  
36      @SuppressWarnings("all")
37      public BearerTokenAuthorizer(final String token) {
38          this.token = token;
39      }
40  }
