8   package pl.chilldev.commons.client.codec;
9   
10  import java.io.IOException;
11  import java.lang.reflect.Type;
12  import java.util.Collections;
13  import java.util.HashMap;
14  import java.util.Map;
15  
16  import feign.FeignException;
17  import feign.Response;
18  import feign.codec.Decoder;
19  import org.springframework.http.HttpHeaders;
20  
21  /**
22   * Content-Type-sensitive decoder.
23   */
24  public class DelegateDecoder implements Decoder
25  {
26      /**
27       * Fallback decoder for not-supported types.
28       */
29      private Decoder fallback;
30  
31      /**
32       * Content-Type mapping for custom delegated decoders.
33       */
34      private Map<String, Decoder> delegates = new HashMap<>();
35  
36      /**
37       * Initializes decoder.
38       *
39       * @param fallback Default fallback decoder.
40       */
41      public DelegateDecoder(Decoder fallback)
42      {
43          this.fallback = fallback;
44      }
45  
46      /**
47       * Registers handler for given MIME type.
48       *
49       * @param mimeType MIME content type.
50       * @param decoder Type handler.
51       * @return Self instance.
52       */
53      public DelegateDecoder registerTypeDecoder(String mimeType, Decoder decoder)
54      {
55          this.delegates.put(mimeType, decoder);
56  
57          return this;
58      }
59  
60      /**
61       * {@inheritDoc}
62       */
63      @Override
64      public Object decode(Response response, Type type) throws IOException, FeignException
65      {
66          return this.delegates.getOrDefault(
67              response.headers().getOrDefault(HttpHeaders.CONTENT_TYPE, Collections.emptyList())
68                  .stream()
69                  .findFirst()
70                  .orElse("")
71                  .split(";")[0],
72              this.fallback
73          )
74              .decode(response, type);
75      }
76  }
