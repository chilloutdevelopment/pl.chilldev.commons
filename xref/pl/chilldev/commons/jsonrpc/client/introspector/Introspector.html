37  /**
38   * Introspector for clients classes to automatically map method to JSON calls.
39   */
40  public class Introspector
41  {
42      /**
43       * Wrapper for parameter mapper that reduces dependency to just call parameters.
44       *
45       * @param <Type> Mapping parameter type.
46       */
47      @FunctionalInterface
48      interface ParameterMapperWrapper<Type>
49      {
50          /**
51           * Populates call parameters with given value.
52           *
53           * @param value Parameter value passed to the call.
54           * @param params Current state of RPC call parameters.
55           */
56          void putParam(Type value, Map<String, Object> params);
57      }
58  
59      /**
60       * RPC method call.
61       */
62      static class Call
63      {
64          /**
65           * RPC method name.
66           */
67          private String name;
68  
69          /**
70           * Parameters mappers.
71           */
72          private List<Introspector.ParameterMapperWrapper<Object>> params;
73  
74          /**
75           * Response handler.
76           */
77          private Function<Object, ?> handler;
78  
79          /**
80           * Initializes RPC call handler.
81           *
82           * @param name Request method name.
83           * @param params Parameters mappers.
84           * @param handler Response handler.
85           */
86          Call(
87              String name,
88              List<Introspector.ParameterMapperWrapper<Object>> params,
89              Function<Object, ?> handler
90          )
91          {
92              this.name = name;
93              this.params = params;
94              this.handler = handler;
95          }
96  
97          /**
98           * Executes request on given connector.
99           *
100          * @param connector TCP connector.
101          * @param arguments Request parameters.
102          * @return Response result.
103          */
104         public Object execute(Connector connector, Object[] arguments)
105         {
106             Map<String, Object> params = new HashMap<>();
107             for (int i = 0; i < arguments.length; ++i) {
108                 this.params.get(i).putParam(arguments[i], params);
109             }
110 
111             return this.handler.apply(
112                 params.isEmpty()
113                     ? connector.execute(this.name)
114                     : connector.execute(this.name, params)
115             );
116         }
117     }

119     /**
120      * RPC calls wrapper.
121      */
122     public static class Client
123     {
124         /**
125          * TCP connector.
126          */
127         private Connector connector;
128 
129         /**
130          * RPC calls.
131          */
132         private Map<Method, Introspector.Call> calls = new HashMap<>();
133 
134         /**
135          * Initializes service over given client.
136          *
137          * @param connector TCP connector.
138          */
139         Client(Connector connector)
140         {
141             this.connector = connector;
142         }
143 
144         /**
145          * Registers method call handler.
146          *
147          * @param method Client class method.
148          * @param call RPC call handler.
149          */
150         public void register(Method method, Introspector.Call call)
151         {
152             this.calls.put(method, call);
153         }
154 
155         /**
156          * Executes the RPC call.
157          *
158          * @param method Invoked method.
159          * @param arguments Call-time arguments.
160          * @return Execution result.
161          */
162         @RuntimeType
163         public Object execute(@Origin Method method, @AllArguments Object[] arguments)
164         {
165             return this.calls.get(method).execute(this.connector, arguments);
166         }
167     }

174     /**
175      * Default parameter mapper.
176      */
177     private static final ParameterMapper<Object> DEFAULT_MAPPER
178         = (String name, Object value, Map<String, Object> params) -> params.put(name, value);
179 
180     /**
181      * Transparent response handler.
182      */
183     private static final Function<Object, Object> IDENTITY_HANDLER = (Object value) -> value;
184 
185     /**
186      * Logger.
187      */
188     private Logger logger = LoggerFactory.getLogger(Introspector.class);
189 
190     /**
191      * Parameters mappers.
192      */
193     private Map<Class<?>, ParameterMapper<Object>> mappers = new HashMap<>();
194 
195     /**
196      * Results handlers.
197      */
198     private Map<Class<?>, Function<Object, ?>> handlers = new HashMap<>();
199 
200     /**
201      * Registers parameter mapper for given class.
202      *
203      * @param type Parameter type.
204      * @param mapper Parameter mapper.
205      * @param <Type> Parameter type.
206      */
207     public <Type> void registerParameterMapper(
208         Class<Type> type,
209         ParameterMapper<? super Type> mapper
210     )
211     {
212         this.mappers.put(
213             type,
214             (String name, Object value, Map<String, Object> params)
215                 -> mapper.putParam(name, type.cast(value), params)
216         );
217     }

219     /**
220      * Registers response type handler for given class.
221      *
222      * @param type Response type.
223      * @param handler Response handler.
224      * @param <Type> Response object type.
225      */
226     public <Type> void registerResultHandler(
227         Class<Type> type,
228         Function<Object, ? extends Type> handler
229     )
230     {
231         this.handlers.put(
232             type,
233             handler
234         );
235     }

237     /**
238      * Builds client wrapper that handles all of the JSON-RPC calls.
239      *
240      * @param type Base client type.
241      * @param connector TPC connector to use for the calls.
242      * @param <Type> Client type.
243      * @return Client instance.
244      */
245     public <Type> Class<? extends Type> createClient(Class<Type> type, Connector connector)
246     {
247         return new ByteBuddy()
248             .subclass(type)
249             .method(ElementMatchers.isAnnotatedWith(JsonRpcCall.class))
250             .intercept(MethodDelegation.to(this.buildClient(type, connector)))
251             .make()
252             .load(this.getClass().getClassLoader(), ClassLoadingStrategy.Default.WRAPPER)
253             .getLoaded();
254     }
255 
256     /**
257      * Creates RPC client.
258      *
259      * @param type Base client type.
260      * @param connector TPC connector to use for the calls.
261      * @return RPC client.
262      */
263     private Introspector.Client buildClient(Class<?> type, Connector connector)
264     {
265         Introspector.Client client = new Introspector.Client(connector);
266 
267         for (Method method : type.getMethods()) {
268             if (method.isAnnotationPresent(JsonRpcCall.class)) {
269                 this.logger.debug("Found {}.{} method as JSON-RPC request.", type.getName(), method.getName());
270 
271                 JsonRpcCall call = method.getAnnotation(JsonRpcCall.class);
272 
273                 Parameter[] parameters = method.getParameters();
274                 List<Introspector.ParameterMapperWrapper<Object>> mappers
275                     = new ArrayList<>(parameters.length);
276 
277                 // build parameters resolvers
278                 for (Parameter parameter : parameters) {
279                     // register synthetic RPC handler
280                     mappers.add(this.createParameterMapper(parameter));
281                 }
282 
283                 // response handler
284                 Class<?> response = method.getReturnType();
285                 Function<Object, ?> handler = this.handlers.containsKey(response)
286                     ? this.handlers.get(response)
287                     : Introspector.IDENTITY_HANDLER;
288 
289                 client.register(
290                     // use overridden name if set
291                     method,
292                     new Introspector.Call(
293                         "".equals(call.name()) ? method.getName() : call.name(),
294                         mappers,
295                         handler
296                     )
297                 );
298             }
299         }
300 
301         return client;
302     }

304     /**
305      * Creates mapper for given parameter.
306      *
307      * @param parameter Method parameter.
308      * @return Parameter provider.
309      */
310     private Introspector.ParameterMapperWrapper<Object> createParameterMapper(Parameter parameter)
311     {
312         // try to fetch provider by parameter type
313         Class<?> type = parameter.getType();
314         ParameterMapper<Object> mapper = this.mappers.containsKey(type)
315             ? this.mappers.get(type)
316             : Introspector.DEFAULT_MAPPER;
317 
318         String name = parameter.getName();
319 
320         // override defaults if annotation is defined
321         JsonRpcParam metadata = parameter.getAnnotation(JsonRpcParam.class);
322         if (metadata != null) {
323             name = "".equals(metadata.name()) ? name : metadata.name();
324         }
325 
326         return this.createParameterMapperWrapper(mapper, name);
327     }
328 
329     /**
330      * Creates parameter mapper wrapper for given parameter scope.
331      *
332      * @param mapper Value mapper.
333      * @param name Parameter name.
334      * @param <Type> Parameter type.
335      * @return Wrapped parameter mapper.
336      */
337     private <Type> Introspector.ParameterMapperWrapper<Type> createParameterMapperWrapper(
338         ParameterMapper<? super Type> mapper,
339         String name
340     )
341     {
342         return (Type value, Map<String, Object> params) -> mapper.putParam(name, value, params);
343     }

345     /**
346      * Creates introspector with handlers for common types.
347      *
348      * @return Introspector.
349      */
350     private static Introspector createDefault()
351     {
352         Introspector introspector = new Introspector();
353 
354         // parameters mappers
355 
356         // Spring Data page request mapper
357         introspector.registerParameterMapper(
358             Pageable.class,
359             (String name, Pageable value, Map<String, Object> params) -> {
360                 params.put("page", value.getPageNumber());
361                 params.put("limit", value.getPageSize());
362                 params.put("sort", value.getSort());
363             }
364         );
365 
366         // response types handlers
367 
368         // UUID handling
369         introspector.registerResultHandler(
370             UUID.class,
371             (Object response) -> UUID.fromString(response.toString())
372         );
373 
374         return introspector;
375     }
