8   package pl.chilldev.commons.jsonrpc.mina;
9   
10  import java.nio.charset.StandardCharsets;
11  
12  import org.apache.mina.core.service.IoHandler;
13  import org.apache.mina.core.service.IoService;
14  import org.apache.mina.filter.codec.ProtocolCodecFilter;
15  import org.apache.mina.filter.codec.textline.TextLineCodecFactory;
16  
17  /**
18   * Utility class for Mina I/O services.
19   */
20  public class IoServiceUtils
21  {
22      /**
23       * Interface for I/O services configuration.
24       */
25      public interface Configuration
26      {
27          /**
28           * Returns maximum packet size.
29           *
30           * @return Maximum packet size.
31           */
32          int getMaxPacketSize();
33      }
34  
35      /**
36       * Key for codec chains.
37       */
38      private static final String CHAIN_CODEC = "codec";
39  
40      /**
41       * Initializes I/O service for text protocols.
42       *
43       * @param service I/O service connector.
44       * @param handler Protocol handler.
45       * @param config Configuration.
46       */
47      public static void initialize(IoService service, IoHandler handler, IoServiceUtils.Configuration config)
48      {
49          // text codec settings
50          TextLineCodecFactory codec = new TextLineCodecFactory(StandardCharsets.UTF_8);
51          codec.setDecoderMaxLineLength(config.getMaxPacketSize());
52          codec.setEncoderMaxLineLength(config.getMaxPacketSize());
53  
54          // network service configuration
55          service.getFilterChain().addLast(IoServiceUtils.CHAIN_CODEC, new ProtocolCodecFilter(codec));
56          service.setHandler(handler);
57      }
58  }
