8   package pl.chilldev.commons.jsonrpc.json.writer;
9   
10  import java.io.IOException;
11  
12  import java.util.ArrayList;
13  import java.util.List;
14  
15  import net.minidev.json.JSONStyle;
16  import net.minidev.json.JSONValue;
17  import net.minidev.json.reader.JsonWriterI;
18  
19  import org.springframework.data.domain.Sort;
20  
21  /**
22   * Spring-Data Sort object serializer.
23   */
24  public class SortWriter
25      implements
26          JsonWriterI<Sort>
27  {
28      /**
29       * Dumps object into JSON-serializable array and writes it as a JSON string.
30       *
31       * @param value Value.
32       * @param out Output stream.
33       * @param compression JSON formatter style.
34       * @throws IOException When write to output stream fails.
35       */
36      @Override
37      public void writeJSONString(Sort value, Appendable out, JSONStyle compression)
38          throws
39              IOException
40      {
41          List<List<String>> sort = new ArrayList<>();
42  
43          // serialize each sorting property into two-element array of form [property, direction]
44          if (value != null) {
45              for (Sort.Order single : value) {
46                  List<String> order = new ArrayList<>();
47                  order.add(single.getProperty());
48                  order.add(single.getDirection().toString());
49                  sort.add(order);
50              }
51          }
52  
53          JSONValue.writeJSONString(sort, out, compression);
54      }
55  }
