8   package pl.chilldev.commons.jsonrpc.rpc;
9   
10  import com.thetransactioncompany.jsonrpc2.JSONRPC2Error;
11  import com.thetransactioncompany.jsonrpc2.JSONRPC2Request;
12  import com.thetransactioncompany.jsonrpc2.JSONRPC2Response;
13  
14  import pl.chilldev.commons.jsonrpc.daemon.ContextInterface;
15  
16  /**
17   * Non-result RPC method wrapper.
18   *
19   * @param <ContextType> Service request context type (will be used as context for request handlers).
20   */
21  @FunctionalInterface
22  public interface VoidMethod<ContextType extends ContextInterface>
23  {
24      /**
25       * JSON-RPC call handler.
26       *
27       * @param <ContextType> Service request context type.
28       */
29      class RequestHandler<ContextType extends ContextInterface>
30          implements
31              Dispatcher.RequestHandler<ContextType>
32      {
33          /**
34           * RPC method handler.
35           */
36          private VoidMethod<? super ContextType> method;
37  
38          /**
39           * Initializes method wrapper.
40           *
41           * @param method RPC method.
42           */
43          public RequestHandler(VoidMethod<? super ContextType> method)
44          {
45              this.method = method;
46          }
47  
48          /**
49           * {@inheritDoc}
50           */
51          @Override
52          public JSONRPC2Response process(JSONRPC2Request request, ContextType context)
53              throws
54                  JSONRPC2Error
55          {
56              this.method.process(request, context);
57              return new JSONRPC2Response(request.getID());
58          }
59      }
60  
61      /**
62       * Handles method call.
63       *
64       * @param request Request call.
65       * @param context Execution context.
66       * @throws JSONRPC2Error Error during JSON-RPC call handling.
67       */
68      void process(JSONRPC2Request request, ContextType context)
69          throws
70              JSONRPC2Error;
71  }
