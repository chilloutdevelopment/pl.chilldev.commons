8   package pl.chilldev.commons.db.timestampable;
9   
10  import java.lang.reflect.Field;
11  
12  import java.time.OffsetDateTime;
13  
14  import javax.persistence.PrePersist;
15  import javax.persistence.PreUpdate;
16  
17  import org.apache.commons.lang3.ArrayUtils;
18  import org.apache.commons.lang3.reflect.FieldUtils;
19  
20  /**
21   * Handler for timestampable annotations.
22   */
23  public class TimestampableListener
24  {
25      /**
26       * Fills all marked fields with timestamps.
27       *
28       * @param timestampable Object to be filled.
29       * @param timestampType Type of operation to handler.
30       * @throws IllegalAccessException When accessing source field is impossible.
31       */
32      private void generateTimestamps(Object timestampable, TimestampType timestampType)
33          throws
34              IllegalAccessException
35      {
36          Class<?> type = timestampable.getClass();
37  
38          for (Field field : FieldUtils.getAllFields(type)) {
39              if (field.isAnnotationPresent(Timestamp.class)) {
40                  // get timestamp configuration
41                  Timestamp options = field.getDeclaredAnnotation(Timestamp.class);
42  
43                  // verify timestamp type
44                  if (ArrayUtils.contains(options.value(), timestampType)) {
45                      // generate timestamp
46                      FieldUtils.writeField(timestampable, field.getName(), OffsetDateTime.now(), true);
47                  }
48              }
49          }
50      }
51  
52      /**
53       * Saves timestamps marked for entity creation creation.
54       *
55       * @param timestampable Subject for timestamp handling.
56       * @throws IllegalAccessException When accessing source field is impossible.
57       */
58      @PrePersist
59      public void createTimestamps(Object timestampable)
60          throws
61              IllegalAccessException
62      {
63          this.generateTimestamps(timestampable, TimestampType.CREATE);
64      }
65  
66      /**
67       * Saves timestamps marked for entity update creation.
68       *
69       * @param timestampable Subject for timestamp handling.
70       * @throws IllegalAccessException When accessing source field is impossible.
71       */
72      @PreUpdate
73      public void updateTimestamps(Object timestampable)
74          throws
75              IllegalAccessException
76      {
77          this.generateTimestamps(timestampable, TimestampType.UPDATE);
78      }
79  }
