8   package test.pl.chilldev.commons.text.html;
9   
10  import org.junit.Test;
11  import static org.junit.Assert.*;
12  
13  import pl.chilldev.commons.text.html.Utils;
14  
15  public class UtilsTest
16  {
17      @Test
18      public void firstParagraph()
19      {
20          // just for code coverage
21          new Utils();
22  
23          String value = "foo <span>bar</span>\n<em>baz</em>";
24          String html = "<p>" + value + "</p> <p>\nquux</p>";
25  
26          assertEquals(
27              "Utils.firstParagraph() should return content of the first paragraph from given snippet.",
28              value,
29              Utils.firstParagraph(html)
30          );
31      }
32  
33      @Test
34      public void firstParagraphEmpty()
35      {
36          String value = "foo <span>bar</span>\n<em>baz</em>";
37  
38          assertEquals(
39              "Utils.firstParagraph() should return empty string if there is no paragraph in given snippet.",
40              "",
41              Utils.firstParagraph(value)
42          );
43      }
44  }
