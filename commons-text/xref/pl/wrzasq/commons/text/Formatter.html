8   package pl.wrzasq.commons.text;
9   
10  import java.util.HashMap;
11  import java.util.Map;
12  
13  import pl.wrzasq.commons.text.formatter.FormatterInterface;
14  
15  /**
16   * Text formats handler.
17   */
18  public class Formatter {
19      /**
20       * All known format handlers.
21       */
22      private Map<String, FormatterInterface> formatters = new HashMap<>();
23  
24      /**
25       * Registers format handler.
26       *
27       * @param format Format name.
28       * @param handler Format handler.
29       */
30      public void registerFormatter(String format, FormatterInterface handler) {
31          this.formatters.put(format, handler);
32      }
33  
34      /**
35       * Transforms text from given form.
36       *
37       * @param format Format name.
38       * @param text Source text.
39       * @return Generated markup snippet.
40       * @throws TextProcessingException When specified format can't be handled or handler throws exception.
41       */
42      public String transform(String format, String text)
43          throws
44              TextProcessingException {
45          if (!this.formatters.containsKey(format)) {
46              throw new TextProcessingException(String.format("No handler for format \"%s\" registred.", format));
47          }
48  
49          return this.formatters.get(format).transform(text);
50      }
51  }
